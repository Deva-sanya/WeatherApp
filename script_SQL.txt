DROP TABLE IF EXISTS weather;
CREATE TABLE IF NOT EXISTS weather
( id integer, 
 Temperature_Celsius decimal(2, 1),
  Wind_MPH decimal(2, 1),
  Pressure_MB decimal(5, 1),
  Humidity INT,
  Condition varchar (256),
  Location varchar (256),
  Time_Of_Last_Update timestamp(2)
);
